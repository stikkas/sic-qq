/**
 * Заголовочная панель с кнопками управления.
 */
Ext.define('qqext.view.VTitleBar', {
	extend: 'Ext.panel.Panel',
	requires: [
		'qqext.Menu'
	],
	title: 'АС Запросы',
	region: 'north',
	layout: 'hbox',
	margin: '0 0 10 0',
	initComponent: function() {
		var me = this,
				menus = qqext.Menu;
		Ext.applyIf(me, {
			items: [menus.editMenu, menus.navigation]});
		me.callParent(arguments);
	}
});
